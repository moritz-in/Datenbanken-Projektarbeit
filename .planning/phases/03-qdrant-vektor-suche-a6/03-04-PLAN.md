    The single `search()` function handles GET and POST for all 6 tab types.

    ```python
    @bp.route("/search", methods=["GET", "POST"])
    def search():
        search_type = request.args.get("type") or request.form.get("type") or "vector"
        query = request.form.get("query", "").strip() if request.method == "POST" else request.args.get("query", "")
        topk = _get_int(request.form.get("topk"), 5)
        results = []
        answer = None

        if request.method == "POST" and query:
            try:
                search_svc = ServiceFactory.get_search_service()
                prod_svc = ServiceFactory.get_product_service()

                if search_type == "vector":
                    results = search_svc.vector_search(query, topk=topk)
                    if not results:
                        flash("Qdrant-Index leer — bitte zuerst Index aufbauen unter /index", "warning")

                elif search_type == "sql":
                    results = prod_svc.execute_sql_query(query)

                elif search_type in ("rag", "graph", "pdf", "pdf_mgmt"):
                    # Phase 4 methods — catch NotImplementedError, return empty
                    try:
                        if search_type == "rag":
                            result = search_svc.rag_search("C", query, topk=topk)
                            results = result.get("hits", [])
                            answer = result.get("answer")
                        elif search_type == "graph":
                            result = search_svc.rag_search("C", query, topk=topk, use_graph_enrichment=True)
                            results = result.get("hits", [])
                            answer = result.get("answer")
                        elif search_type == "pdf":
                            result = search_svc.pdf_rag_search(query, topk=topk)
                            results = (result or {}).get("hits", [])
                            answer = (result or {}).get("answer")
                        elif search_type == "pdf_mgmt":
                            results = search_svc.search_product_pdfs(query, topk=topk)
                    except NotImplementedError:
                        results = []
                        answer = None

            except ValueError as e:
                flash(str(e), "danger")
            except Exception as e:
                log.exception("Search error: %s", e)
                flash(f"Suchfehler: {e}", "danger")

        return render_template(
            "search_unified.html",
            search_type=search_type,
            query=query,
            results=results,
            answer=answer,
        )
    ```

    Key rules:
    - Default `search_type` when no param: `"vector"` (per CONTEXT.md locked decision)
    - Phase 4 methods (rag, graph, pdf, pdf_mgmt) must catch `NotImplementedError` and return empty results — NOT 501
    - ValueError from execute_sql_query (invalid SQL) flashes with "danger" — does not 500
    - Use `flash(message, "warning")` for empty Qdrant index (not "danger")
